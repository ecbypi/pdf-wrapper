509       raise ArgumentError, "file #{filename} not found" unless File.file?(filename)
510 
511       filetype = detect_image_type(filename)
512 
513       if filetype.eql?(:png)
514         img_surface = Cairo::ImageSurface.from_png(filename)
515         x, y = current_point
516         @context.set_source(img_surface, opts[:left] || x, opts[:top] || y)
517         @context.paint
518       elsif filetype.eql?(:svg)
519         # thanks to Nathan Stitt for help with this section
520         load_librsvg

530           end
531         end

594     def build_pango_layout(str, opts = {})
595       options = {:left => @margin_left,
596                  :top => @margin_top,
597                  :font => @default_font,
598                  :font_size => @default_font_size,
599                  :color => @default_color,
600                  :alignment => :left,
601                  :justify => false,
602                  :spacing => 0
603                 }
604       options.merge!(opts)
605 
606       # if the user hasn't specified a width, make the text wrap on the right margin
607       options[:width] = absolute_right_margin - options[:left] if options[:width].nil?
608 
609       # even though this is a private function, raise this error to force calling functions
610       # to decide how they want to handle converting non-strings into strings for rendering
611       raise ArgumentError, 'build_pango_layout must be passed a string' unless str.kind_of?(String)
612 
613       # if we're running under a M17n aware VM, ensure the string provided is UTF-8
614       if RUBY_VERSION >= "1.9"
615         begin
616           str = str.encode("UTF-8")
617         rescue
618           raise ArgumentError, 'Strings must be supplied with a UTF-8 encoding, or an encoding that can be converted to UTF-8'
619         end 
620       end
621       
622       # The pango way:
623       load_libpango
624 
625       # create a new Pango layout that our text will be added to
626       layout = @context.create_pango_layout
627       layout.text = str.to_s
628       layout.width = options[:width] * Pango::SCALE
629       layout.spacing = options[:spacing] * Pango::SCALE
630 
631       # set the alignment of the text in the layout
632       if options[:alignment].eql?(:left)
633         layout.alignment = Pango::Layout::ALIGN_LEFT
634       elsif options[:alignment].eql?(:right)
635         layout.alignment = Pango::Layout::ALIGN_RIGHT
636       elsif options[:alignment].eql?(:center) || options[:alignment].eql?(:centre)
637         layout.alignment = Pango::Layout::ALIGN_CENTER
638       else
639         raise ArgumentError, "Invalid alignment requested"
640       end
641 
642       # justify the text if need be - only works in pango >= 1.17
643       layout.justify = true if options[:justify]
644 
645       # setup the font that will be used to render the text
646       fdesc = Pango::FontDescription.new(options[:font])
647       fdesc.set_size(options[:font_size] * Pango::SCALE)
648       layout.font_description = fdesc
649       @context.update_pango_layout(layout)
650       return layout
651     end

679     # adds a single table row to the canvas. Top left of the row will be at the current x,y
680     # co-ordinates, so make sure they're set correctly before calling this function
681     #
682     # strings - array of strings. Each element of the array is a cell
683     # column_widths - the width of each column. At this stage it should be an int. All columns are the same width
684     # options - any options relating to text style to use. font, font_size, alignment, etc. See text() for more info.
685     #
686     # Returns the y co-ordinates of the bottom edge of the row, ready for the next row
687     def draw_table_row(strings, column_widths, options)
688       row_height = 0
689       x, y = current_point
690 
691       # we run all this code twice. The first time is a dry run to calculate the
692       # height of the largest cell, which determines the overall height of the row.
693       # The second run through we actually draw each cell onto the canvas
694       [:dry, :paint].each do |action|
695 
696         strings.each do |head|
697           # TODO: provide a way for these to be overridden on a per cell basis
698           opts = {
699             :font      => options[:font],
700             :font_size => options[:font_size],
701             :color     => options[:color],
702             :alignment => options[:alignment],
703             :justify   => options[:justify],
704             :spacing   => options[:spacing]
705           }
706 
707           if action == :dry
708             # calc the cell height, and set row_height if this cell is the biggest in the row
709             cell_height = text_height(head, column_widths, opts)
710             row_height = cell_height if cell_height > row_height
711           else
712             # start a new page if necesary
713             if row_height > (absolute_bottom_margin - y)
714               start_new_page 
715               y = margin_top
716             end
717 
718             # add our cell, then advance x to the left edge of the next cell
719             self.cell(head, x, y, column_widths, row_height, opts)
720             x += column_widths 
721           end
722 
723         end
724       end
725 
726       return y + row_height
727     end

748     # renders a pango layout onto our main context
749     # based on a function of the same name found in the text2.rb sample file 
750     # distributed with rcairo - it's still black magic to me
751     def render_layout(layout, x, y, h, opts = {})
752       options = {:auto_new_page => true }
753       options.merge!(opts)
754 
755       limit_y = y + h
756 
757       iter = layout.iter
758       prev_baseline = iter.baseline / Pango::SCALE
759       begin
760         line = iter.line
761         ink_rect, logical_rect = iter.line_extents
762         y_begin, y_end = iter.line_yrange
763         if limit_y < (y + y_end / Pango::SCALE)
764           if options[:auto_new_page]
765             start_new_page
766             y = margin_top - prev_baseline
767           else
768             break
769           end
770         end
771         width, height = layout.size
772         baseline = iter.baseline / Pango::SCALE
773         move_to(x + logical_rect.x / Pango::SCALE, y + baseline)
774         @context.show_pango_layout_line(line)
775         prev_baseline = baseline
776       end while iter.next_line!
777       return y + baseline
778     end
